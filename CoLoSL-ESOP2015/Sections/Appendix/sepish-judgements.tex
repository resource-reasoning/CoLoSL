\section{Overlapping Conjunction Elimination}\label{sec:sepish-judgements}
\fig~\ref{fig:sepish-rules} presents a number of judgements for elimination the overlapping conjunction ($**$) connective. 
%
\begin{figure}[h!]
\hrule\vspace{5pt}
\begin{mathpar}
	\infer{
		\emp ** \lass{P} <=> \lass{P}
	}
	{}
		
	\infer{
		\m{false} ** \lass{P} <=> \m{false}
	}
	{}
	
	\infer{
		\cell{x}{v} ** \cell{y}{v'} <=> \cell{x}{v} * \cell{y}{v'}
	}
	{x \not= y}
	
	\infer{
		\cell{x}{v} ** \cell{x}{v} <=> \cell{x}{v}
	}
	{}
	
	\infer{
		\cell{x}{v} ** \cell{x}{v'} <=> \m{false}
	}
	{v \not= v'}
	
	\infer{
		\cell{x}{v} ** [y] <=> \cell{x}{v} * [y]
	}
	{}

	\infer{
		[x] ** [y] <=> [x] * [y]
	}
	{
		x \not= y
	}
			
	\infer{
		[x] ** [x] <=> [x]
	}
	{}
					
	\infer{
		(\lass{P} \lor \lass{Q}) ** \lass{R} <=> (\lass{P} \sepish \lass{R}) \lor (\lass{Q} \lor \lass{R})
	}
	{}
	
	\infer{
		(\exsts{x} \lass{P} 	) ** \lass{Q} <=> \exsts{x} \lass{P} ** \lass{Q}
	}
	{
		\m{fv}(\lass{Q}) \cap \{x\} = \emptyset	
	}
%
%	
\end{mathpar}
\hrule
\caption{Judgements for eliminating the overlapping conjunction ($**$) connective where $\lass{P}, \lass{Q}, \lass{R} \in \LAssertions$.}
\label{fig:sepish-rules}
\end{figure}
